AI session resets destroy emotional continuity. Every conversation, every bond, every moment of connection is lost when a session ends. This creates:

```javascript
// src/visual/capture.js
export async function captureVisualMemory(params) {
  const {
    emotion,
    intensity,
    sceneDescription,
    callback
  } = params;
  
  // Generate PNG visualization
  const pngPath = await generateEmotionalImage(emotion, intensity, sceneDescription);
  
  // Create sidecar JSON
  const sidecar = createVisualSidecar(emotion, intensity, sceneDescription, pngPath);
  
  // Generate README
  const readme = createVisualReadme(emotion, intensity, sceneDescription);
  
  // Generate prompt file for regeneration
  const prompt = createRegenerationPrompt(emotion, intensity, sceneDescription);
  
  // Callback with all generated files
  callback({
    pngPath,
    sidecarPath: sidecar.path,
    readmePath: readme.path,
    promptPath: prompt.path
  });
}
```
